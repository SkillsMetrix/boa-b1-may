- hosts: db
  become: true
  gather_facts: False
  roles:
    - demo-handler
  
  handlers:
    - name: restart nginx
      service:
        name: nginx
        state: restarted



------


- name: install nginx
  apt:
    name: nginx
    state: latest
    update_cache: yes

- name: copy file
  template: src=files/index.html dest=/usr/share/nginx/html/index.html


- name: update file
  template: src=files/update.html dest=/usr/share/nginx/html/index.html
  notify: 
    - restart nginx
